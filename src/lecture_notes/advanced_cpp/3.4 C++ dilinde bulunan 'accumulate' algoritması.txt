                ivec.begin(), 
                ivec.end(),

                [](int prev_result, int val) { return prev_result + val; }
            ) << '\n'; // 25

                ivec.begin(), 
                ivec.end(), 
                ""s, 
                [](string prev_result, int val) { return prev_result + '-' + to_string(val); }
